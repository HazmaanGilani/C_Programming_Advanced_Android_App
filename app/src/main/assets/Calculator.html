/*Simple calculator using switch case.*/ 

#include<stdio.h>
int main()
{
int a,b,n,s,m,su,d;
printf("enter two no. : ");
scanf("%d%d",&a,&b);
printf("enter 1 for sum, 2 for multiply, 3 for subtraction 4 for division:");
scanf("%d",&n);
switch(n)
{
case 1:
s=a+b;
printf("sum=%d",s);
break;
case 2:
m=a*b;
printf("multiply=%d",m);
break;
case 3:
su=a-b;
printf("subtraction=%d",su);
break;
case 4:
d=a/b;
printf("divission=%d",d);
break; 
default:
printf("wrong input");
break;
}
}
