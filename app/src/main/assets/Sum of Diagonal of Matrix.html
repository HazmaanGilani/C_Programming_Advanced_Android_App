/*Sum of diagonal elements of a matrix */ 

#include<stdio.h>
void main(){
  int a[10][10],i,j,sum=0,m,n;
  printf("\nEnter the no of rows and columns of matrix: ");
  scanf("%d%d",&m,&n);
  printf("\nEnter the elements of matrix: ");
  for(i=0;i<m;i++)
      for(j=0;j<n;j++)
           scanf("%d",&a[i][j]);
  printf("\nThe matrix is\n");
  for(i=0;i<m;i++){
      printf("\n");
      for(j=0;j<m;j++){
      printf("%d\t",a[i][j]);
      }
 }
 for(i=0;i<m;i++){
     for(j=0;j<n;j++){
          if(i==j)
              sum=sum+a[i][j];
     }
 }
 printf("\nSum of the diagonal elements of a matrix are: %d",sum);
}
