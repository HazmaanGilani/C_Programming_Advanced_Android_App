void shell_sort(int a[], int total, int index) 

{
int i = 0; 
int j = 0; 
int k = 0; 
int l = 0;

for(k = 0; k < index; k++) 
{

 for(i = k; i < total; i += index) 

{

  l = a[i];
  for(j = (i - index); j >= 0; j -= index) 

{
  if(a[j] > l)

   a[j + index] = a[j];

   else
    break;

  }

  a[j + index] = l;

 }

}

 return 0;

}
