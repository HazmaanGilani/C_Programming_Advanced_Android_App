/*C Program to read data from File*/ 

#include <stdio.h>
#include<stdlib.h>
int main()
{
   char ch[100];
   FILE *fptr;

   if ((fptr=fopen("file1.txt","r"))==NULL){
       printf("Error! opening file");
       exit(0);         
   }

   fscanf(fptr,"%[^\n]",ch);
   printf("Data from file:\n%s",ch);
   fclose(fptr);
   return 0;
}
